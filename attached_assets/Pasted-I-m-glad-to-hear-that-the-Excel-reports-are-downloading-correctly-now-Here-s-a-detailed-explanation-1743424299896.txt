Main Fixes for Excel Download Issue
1. Direct Streaming Instead of Temporary Files
The most important change was replacing the temporary file creation approach with direct streaming. Previously, the code was:

Creating an Excel workbook
Writing it to a temporary file on disk
Then serving that file to the client
Trying to clean up the file afterward
This approach had several problems:

File permissions issues
Cleanup failures leaving orphaned files
Potential path issues across different operating systems
The fix I implemented uses direct streaming to the response:

// Set correct MIME type headers for Excel
res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
res.setHeader('Content-Disposition', `attachment; filename="${type}-report.xlsx"`);
          
// Write directly to the response stream to avoid file system operations
await workbook.xlsx.write(res);

else if (format === 'xlsx') {
  // Generate Excel file with report data
  try {
    const workbook = new Excel.Workbook();
    const dataSheet = workbook.addWorksheet('Report Data');
    const summarySheet = workbook.addWorksheet('Summary');
    
    // Make sure worksheets are properly configured
    dataSheet.properties.defaultColWidth = 20;
    summarySheet.properties.defaultColWidth = 25;
    
    // Add title row with formatting
    const reportTitle = `${type.charAt(0).toUpperCase() + type.slice(1).replace(/-/g, ' ')} Report`;
    const titleRow = dataSheet.addRow([reportTitle]);
    titleRow.font = { bold: true, size: 16 };
    dataSheet.mergeCells('A1:F1');
    dataSheet.getCell('A1').alignment = { horizontal: 'center' };
    
    const summaryTitleRow = summarySheet.addRow([reportTitle]);
    summaryTitleRow.font = { bold: true, size: 16 };
    summarySheet.mergeCells('A1:B1');
    summarySheet.getCell('A1').alignment = { horizontal: 'center' };
    
    // Add date with formatting
    const currentDate = new Date().toLocaleDateString();
    const dateRow = dataSheet.addRow([`Generated on ${currentDate}`]);
    dateRow.font = { italic: true };
    dataSheet.mergeCells('A2:F2');
    dataSheet.getCell('A2').alignment = { horizontal: 'center' };
    
    const summaryDateRow = summarySheet.addRow([`Generated on ${currentDate}`]);
    summaryDateRow.font = { italic: true };
    summarySheet.mergeCells('A2:B2');
    summarySheet.getCell('A2').alignment = { horizontal: 'center' };
    
    // Add a spacer row
    dataSheet.addRow([]);
    summarySheet.addRow([]);
    
    // Add headers with formatting
    if (reportData.length > 0) {
      // Format headers in data sheet
      const headers = Object.keys(reportData[0]);
      const headerRow = dataSheet.addRow(headers);
      headerRow.font = { bold: true };
      headerRow.eachCell((cell) => {
        cell.fill = {
          type: 'pattern',
          pattern: 'solid',
          fgColor: { argb: 'FFE0E0E0' }
        };
        cell.border = {
          top: { style: 'thin' },
          left: { style: 'thin' },
          bottom: { style: 'thin' },
          right: { style: 'thin' }
        };
        cell.alignment = { horizontal: 'center' };
      });
      
      // Add data rows with formatting
      reportData.forEach((row: any) => {
        const dataRow = dataSheet.addRow(Object.values(row));
        dataRow.eachCell((cell) => {
          cell.border = {
            top: { style: 'thin' },
            left: { style: 'thin' },
            bottom: { style: 'thin' },
            right: { style: 'thin' }
          };
        });
      });
      
      // Auto-size columns based on content
      headers.forEach((_, colIndex) => {
        const col = dataSheet.getColumn(colIndex + 1);
        col.width = 20;
      });
    }
    
    // Add basic summary stats with formatting
    const summaryHeaderRow = summarySheet.addRow(['Report Statistics']);
    summaryHeaderRow.font = { bold: true, size: 14 };
    summarySheet.mergeCells('A5:B5');
    summarySheet.addRow([]);
    
    const metricsHeaderRow = summarySheet.addRow(['Metric', 'Value']);
    metricsHeaderRow.font = { bold: true };
    metricsHeaderRow.eachCell((cell) => {
      cell.fill = {
        type: 'pattern',
        pattern: 'solid',
        fgColor: { argb: 'FFE0E0E0' }
      };
      cell.border = {
        top: { style: 'thin' },
        left: { style: 'thin' },
        bottom: { style: 'thin' },
        right: { style: 'thin' }
      };
      cell.alignment = { horizontal: 'center' };
    });
    
    // Add report stats with formatting
    const keyMetrics = [
      ['Report Type', reportTitle],
      ['Number of Records', reportData.length.toString()],
      ['Generated Date', currentDate]
    ];
    
    keyMetrics.forEach(([metric, value]) => {
      const row = summarySheet.addRow([metric, value]);
      row.eachCell((cell) => {
        cell.border = {
          top: { style: 'thin' },
          left: { style: 'thin' },
          bottom: { style: 'thin' },
          right: { style: 'thin' }
        };
      });
    });
    
    // Set correct MIME type headers for Excel
    res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
    res.setHeader('Content-Disposition', `attachment; filename="${type}-report.xlsx"`);
    
    // Write directly to the response stream to avoid file system operations
    await workbook.xlsx.write(res);
    return;
  } catch (err) {
    console.error("Error creating Excel file:", err);
    return res.status(500).json({ error: "Failed to create Excel report", details: err instanceof Error ? err.message : String(err) });
  }

Using modern Excel formatting standards
